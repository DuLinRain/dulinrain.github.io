使用ES6解构语法进行代码优化

一、概述

本文将介绍ES6解构语法对代码进行优化的方法。具体介绍：

  1. undefined优化
  2. 对象合并优化

二、undefined优化

在JavaScript中进行undefiend优化主要有以下几种：

  1. 取反判断（✘）
  2. 直接比较undefined（✘）
  3. 三元运算符判断（✘）
  4. ||运算符判断（✘）
  5. typeof判断（✔）

其中只有最后一种可以准确地判断出undfined并实现我们期望的目的。至于为什么，可以看一下另外一篇文章。

我们先看一下例子：

2.1、原始方案

//undefined
function processUserInfo (options) {
    let name = options.name
    let age = options.age
    let isMarried = options.isMarried
    //通过if判断
    if (!options.xxx) {
        name = 'linyu'//do something
    }
    // or
    if (options.xxx === undefined) {
        //do something
    }
    // or
    if (typeof options.xxx === "undefined") {
        //do something
    }
}

2.2、通过三元运算符优化

function processUserInfo (options) {
    //通过三元运算符
    let name = options.name ? options.name : 'linyu'
    let age = options.age ? options.age : 20
    let isMarried = options.isMarried ? options.isMarried: true
}

可以看到，代码量变少了，但是任然无法准确判断undefiend

2.3、通过||运算符优化

function processUserInfo (options) {
    //通过||运算符
    let name = options.name || 'linyu'
    let age = options.age || 20
    let isMarried = options.isMarried || true
}

2.4、通过void运算符优化

function processUserInfo (options) {
    //if
    if (options.xxx === void 0) {
        //do something
    }
    //三元运算符
    let xxx = options.name === void 0 ? 'linyu' : options.name
    // && 运算符
    let xxx = options.name
    (xxx === void 0) && (xxx = 'linyu')
}

2.5、通过void 0 + 三元运算符优化

function processUserInfo (options) {
    //三元运算符
    let name = options.name === void 0 ? 'linyu' : options.name
    let age = options.age === void 0 ? 0 : options.age
    let isMarried = options.isMarried === void 0 ? true : options.isMarried
}

2.6、通过void 0 + &&算符优化

//通过void 0 + &&运算符优化
function processUserInfo (options) {
    // && 运算符
    let name = options.name
    let age = options.age
    let isMarried = options.isMarried
    (name === void 0) && (name = 'linyu')
    (age === void 0) && (age = 10)
    (isMarried === void 0) && (isMarried = true)
}

2.7、通过解构 + void优化

//通过解构 + void优化
function processUserInfo (options) {
    let {name, age, isMarried} =options
    //if
    if (xxx === void 0) {
        //do something
    }
    //三元运算符
    xxx = xxx === void 0 ? 'linyu' : xxx
    // && 运算符
    (xxx === void 0) && (xxx = 'linyu')
}

2.8、通过解构 + 三元运算符 + void优化

//通过解构 + 三元运算符 + void优化
function processUserInfo (options) {
    let {name, age, isMarried} =options
    name = name === void 0 ? 'linyu' : name
    age = age === void 0 ? 0 : age
    isMarried = isMarried === void 0 ? true : isMarried
}

2.9、通过解构 + &&运算符 + void优化

function processUserInfo (options) {
    let {name, age, isMarried} =options
    (name === void 0) && (name = 'linyu')
    (age === void 0) && (age = 10)
    (isMarried === void 0) && (isMarried = true)
}

void 0 就完美了吗，有没有更优雅的方法,优化成1条代码?

2.10、通过解构+默认值优化

function processUserInfo (options) {
    let {name = 'linyu', age = 0, isMarried = true} =options
}

解构+默认值就完美了吗，有没有更优雅的方法,优化成0条代码?

三、对象合并优化

假设我们现在有个公共信息对象，需要将它和options的部分信息组成一个新的对象。通常的做法是这样的：

//公共信息
const commonInfo = {
    company: 'BAIDU',
    address: '西二旗'
}
function processUserInfo (options) {
    //拿基本信息
    let name = options.name || 'linyu'
    let age = options.age || 20
    let isMarried = options.isMarried || true

    //合并基本信息和公共信息
    let allInfo = {}
    allInfo.company = commonInfo.company
    allInfo.address = commonInfo.address
    allInfo.name = name
    allInfo.age = age
    allInfo.isMarried = isMarried
}

3.1、解构+默认值优化

function processUserInfo ({name = 'linyu', age = 0, isMarried = true}) {
    //合并基本信息和公共信息
    let allInfo = {}
    allInfo.company = commonInfo.company
    allInfo.address = commonInfo.address
    allInfo.name = name
    allInfo.age = age
    allInfo.isMarried = isMarried
}

可不可以优化为一条代码呢？

3.2、解构+默认值 + 重构优化为一条代码

//解构+默认值 + 重构优化为一条代码
function processUserInfo ({name = 'linyu', age = 0, isMarried = true}) {
    //合并基本信息和公共信息
    let allInfo = {
        company: commonInfo.company,
        address: commonInfo.address,
        name,
        age,
        isMarried
    }
}

这就完美了吗？还能不能优化？

function processUserInfo ({name = 'linyu', age = 0, isMarried = true}) {
    let allInfo = { ...commonInfo, name, age, isMarried }
}

其实Object.assing也可以做到

function processUserInfo (options) {
    let allInfo = Object.assign({
        name = 'linyu',
        age = 0,
        isMarried = true
    }, options, commonInfo)
}

若不考虑默认值，重构可以更简单

function processUserInfo (options) {
    let allInfo = { ...commonInfo, ...options }
}

2.3、其它思考

考虑到

  • 一方面options中的某个属性可能需要单独拎出来用
  • 另一方面options中的属性并不是全都需要合并

所以采用部分解构的方案是最佳的

function processUserInfo ({name = 'linyu', age = 0, isMarried = true}) {
    //合并基本信息和公共信息
    let allInfo = {
        ...commonInfo,
        name,
        age,
        isMarried
    }
    //if need options object
    //if only need part property of commonInfo
}

但这种存在两个缺点：

  1. 后续可能需要访问options对象
  2. commonInfo中也可能不是全部需

因此可以采用这种方式也可以解构后再重构

function processUserInfo (options) {
    let { company } = commonInfo
    let {name = 'linyu', age = 0, isMarried = true} = options
    let allInfo = { company, name, age, isMarried }
}

那么是否存在下面这种看起来完美的方式呢？

function processUserInfo (options) {
    let allInfo = { company, name = 'linyu' } = {...commonInfo, ...options}
}

答案是否定的，company和name都起到了预期效果，但是allInfo挂了，问题出在哪里呢？

我们看看let a = b = c真真的实现是怎样的呢？

实际上的结果是最后一种。

let c = xxx;b=c;a=b

和 let c=xxx;b=c;a=c

都是最后一种效果。这样就能理解为什么allInfo没有name属性了。

四、对象优化实例

我们在用axios请求的时候经常这样写代码：

axios.post(__QUERY_LIST_PROJECT_LIST__.latestversion_buildlog, qs.stringify({
    id
})).then(function (response) {
    if (response.data.statusCode === '200') {
        _.tableData = response.data.projecthistoris || []
        _.total = response.data.num || 0
    } else if (response.data.statusCode === '302') {
        window.location.replace(response.data.url)
    } else {
        _.showError('构建记录拉取失败！失败原因：' + (response.data.msg || '未知错误'))
    }
}).catch(function (error) {
    _.showError(error.message || '未知错误')
})
