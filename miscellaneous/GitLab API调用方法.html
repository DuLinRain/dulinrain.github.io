三、 node客户端

node作为中间客户端的Demo实现如下：

var axios = require('axios')
var qs = require('qs');
var Oauth2 = require('reliable-gitlab-oauth')
var express = require('express'),
    bodyParser = require('body-parser');
var token = ''
var oauth = new Oauth2({
    client_id: 'ca52821d1b097b2239efb944605677311098890a3e1ada13ae2adxxxxxxx',
    client_secret: '826cae07ba0e922cf607da58c99a18e7e76d4b9ee5ebb72a0e11xxxxxx',
    protocol: 'http',
    server_url: 'gitlab.58corp.com'
})
var app = express();
var authorizedURL = oauth.getAuthorizeURL({
    redirect_uri: 'http://10.252.83.144:3000/code'
})
app.use(bodyParser.urlencoded({ extended: true }));

app.use(bodyParser.json());

app.get('/', function (req, res) {
  var body = '<a href="' + authorizedURL + '"> Get Code </a>';
  res.writeHead(200, {
            'Content-Length': body.length,
            'Content-Type': 'text/html; charset=utf-8' });
  res.end(body);
});
app.get('/', function (req, res) {
  var body = '<a href="' + authorizedURL + '"> 确认授权 </a>';
  res.writeHead(200, {
            'Content-Length': body.length,
            'Content-Type': 'text/html; charset=utf-8' });
  res.end(body);
});
app.get('/project-list', function (req, res) {
    var gitlab = require('gitlab')({
        url:   'http://gitlab.58corp.com',
        oauth_token: token
    });
    gitlab.projects.all(function(projects) {
        var result = '<h2>项目列表</h2>'
        result+='<ul>'
        for (var i=0; i<projects.length;i++) {
            result += ('<li>' + projects[i].name + '</li>')
        }
        result+='</ul>'
        res.writeHead(200, {
            'Content-Length': result.length,
            'Content-Type': 'text/html; charset=utf-8' });
        res.end(result);
    });
});
app.get('/code', function (req, res) {
  if (req.query.code) {
        axios.post('http://gitlab.58corp.com/oauth/token', qs.stringify({
          client_id: oauth.client_id,
          client_secret: oauth.client_secret,
          code: req.query.code,
          grant_type: 'authorization_code',
          redirect_uri: 'http://10.252.83.144:3000/code'
        }))
        .then(function (response) {
            token =  response.data.access_token
            res.redirect('/project-list')
        })
        .catch(function (error) {
            console.log(error);
        });
    } else {
        var body = '<a>已授权</a>';
        res.writeHead(200, {
                'Content-Length': body.length,
                'Content-Type': 'text/html; charset=utf-8' });
        res.end(body);
    }

});

// app.use(app.oauth.errorHandler());

app.listen(3000);
